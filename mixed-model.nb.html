predictions = test %>%
  mutate(predicted = predict(model, test, allow.new.levels = TRUE)) %>%

  mutate(observer_class = cut(observer_count, c(-Inf,0, 2^(0:7), Inf)))

Can we do better?

None of the predictor variables correlate with the residuals, suggesting that their explanatory effect is mainly among sites, rather than in the residuals for individual sites.

There also isnt much in the way of consistent autocorrelation in the residuals (correlation of 0.2 explains 4% of the residual variance [100 * 0.2^2], which isnt tremendously useful even if it is statistically significantespecially since the residual variance is only a small portion of the total variance anyway).

resid_matrix = data.frame(site_id = train$site_id, 
                          year = train$year, 
                          residuals = resid(model)) %>% 
  complete(site_id, year) %>% 
  arrange(site_id, year) %>% 
  magrittr::extract2("residuals") %>% 
  matrix(nrow = n_distinct(train$year))
autocorr_matrix = resid_matrix %>% 
  t() %>% 
  cor(use = "pairwise.complete.obs")
plot_upper = function(x, title){
  reshape2::melt(x * upper.tri(x)) %>% 
    ggplot(aes(x = Var1, y = Var2, fill = value)) + 
    geom_tile() +
    scale_fill_gradient2() +
    ggtitle(title)
}
